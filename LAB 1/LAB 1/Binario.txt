/****************************************/
// Encabezado (Libraries)

#define F_CPU 16000000UL

#include <avr/io.h>
#include <avr/interrupt.h>
#include <stdint.h>

#include "display.h"

/****************************************/
// Variables globales y defines

//Entradas botones
#define BOTON_INICIO PC2
#define BOTON_J1 PC0
#define BOTON_J2 PC1

//Salidas leds multiplexados
#define LED_0 PB0
#define LED_1 PB1
#define LED_2 PB2
#define LED_3 PB3
#define TRANSISTOR_J1 PB4
#define TRANSISTOR_J2 PB5

//Meta de la carrera
#define META 10

//Variables para empezar
volatile uint8_t jugadores_habilitados = 0;
volatile uint8_t conteo_activo = 0;
volatile uint8_t numero_conteo = 0;
volatile uint8_t tiempo_timer = 0;

//Guardar conteos
static uint8_t decadas_J1 = 0;
static uint8_t decadas_J2 = 0;

//Estados de botones
static uint8_t estado_J1 = 1;
static uint8_t estado_J2 = 1;
static uint8_t rebote_J1 = 0;
static uint8_t rebote_J2 = 0;

//Control multiplexado
static uint8_t jugador_led = 0;

//Estado de carrera
static uint8_t ganador = 0;

/****************************************/
// Function prototypes

static void puertos(void);
static void inicializar_timer(void);
static uint8_t leer_inicio(void);
static void iniciar_conteo(void);
static void rutina_J1(void);
static void rutina_J2(void);
static void actualizar_leds(void);
static void mostrar_leds(uint8_t numero);
static void revisar_ganador(void);
static void mostrar_ganador(void);

/****************************************/
// Main Function

int main(void)
{
    //Configuracion inicial
    puertos();
    disp_ini();
    display_OFF();
    inicializar_timer();

    //Activa interrupciones
    sei();

    while (1) {
        //Si se apacha empezar la cuenta
        if (leer_inicio() && !conteo_activo) {
            iniciar_conteo();
        }

        //Rutinas separadas de cada jugador
        rutina_J1();
        rutina_J2();

        //Revisar si ya hay ganador
        revisar_ganador();

        //Multiplexar leds de jugadores
        actualizar_leds();
    }
}

/****************************************/
// NON-Interrupt subroutines

static void puertos(void)
{
    //Entradas (Botones)
    DDRC &= (uint8_t)~((1 << BOTON_INICIO) | (1 << BOTON_J1) | (1 << BOTON_J2));
    //Pull-up
    PORTC |= (1 << BOTON_INICIO) | (1 << BOTON_J1) | (1 << BOTON_J2);

    //Salidas de leds y transistores
    DDRB |= (1 << LED_0) | (1 << LED_1) | (1 << LED_2) | (1 << LED_3) | (1 << TRANSISTOR_J1) | (1 << TRANSISTOR_J2);
    PORTB &= (uint8_t)~((1 << LED_0) | (1 << LED_1) | (1 << LED_2) | (1 << LED_3) | (1 << TRANSISTOR_J1) | (1 << TRANSISTOR_J2));
}

static void inicializar_timer(void)
{
    //Timer1 CTC
    TCCR1A = 0;
    TCCR1B = (1 << WGM12);
    //Tiempo de 100 ms
    OCR1A = 6249;
    //Interrupcion timer
    TIMSK1 = (1 << OCIE1A);
    //Prescaler
    TCCR1B |= (1 << CS12);
}

static uint8_t leer_inicio(void)
{
    //Solo detecta una pulsacion
    if ((PINC & (1 << BOTON_INICIO)) == 0) {
        for (volatile uint16_t pausa = 0; pausa < 3000; pausa++) {
        }

        if ((PINC & (1 << BOTON_INICIO)) == 0) {
            while ((PINC & (1 << BOTON_INICIO)) == 0) {
            }
            return 1;
        }
    }

    return 0;
}

static void iniciar_conteo(void)
{
    cli();
    //Reiniciar carrera
    jugadores_habilitados = 0;
    conteo_activo = 1;
    numero_conteo = 5;
    tiempo_timer = 0;
    ganador = 0;
    decadas_J1 = 0;
    decadas_J2 = 0;
    estado_J1 = 1;
    estado_J2 = 1;
    rebote_J1 = 0;
    rebote_J2 = 0;
    jugador_led = 0;
    PORTB &= (uint8_t)~((1 << LED_0) | (1 << LED_1) | (1 << LED_2) | (1 << LED_3) | (1 << TRANSISTOR_J1) | (1 << TRANSISTOR_J2));
    sei();

    mostrar_numero(numero_conteo);
}

static void rutina_J1(void)
{
    uint8_t lectura_actual = 1;

    //No avanzar si no ha terminado el conteo o ya hay ganador
    if (!jugadores_habilitados || ganador != 0) {
        return;
    }

    if ((PINC & (1 << BOTON_J1)) == 0) {
        lectura_actual = 0;
    }

    if (lectura_actual != estado_J1) {
        rebote_J1++;

        if (rebote_J1 >= 5) {
            estado_J1 = lectura_actual;
            rebote_J1 = 0;

            if (estado_J1 == 0) {
                decadas_J1++;
            }
        }
    } else {
        rebote_J1 = 0;
    }
}

static void rutina_J2(void)
{
    uint8_t lectura_actual = 1;

    //No avanzar si no ha terminado el conteo o ya hay ganador
    if (!jugadores_habilitados || ganador != 0) {
        return;
    }

    if ((PINC & (1 << BOTON_J2)) == 0) {
        lectura_actual = 0;
    }

    if (lectura_actual != estado_J2) {
        rebote_J2++;

        if (rebote_J2 >= 5) {
            estado_J2 = lectura_actual;
            rebote_J2 = 0;

            if (estado_J2 == 0) {
                decadas_J2++;
            }
        }
    } else {
        rebote_J2 = 0;
    }
}

static void revisar_ganador(void)
{
    //Revisar si alguien llego primero a la meta
    if (ganador != 0) {
        return;
    }

    if (decadas_J1 >= META) {
        ganador = 1;
        jugadores_habilitados = 0;
        mostrar_ganador();
    } else if (decadas_J2 >= META) {
        ganador = 2;
        jugadores_habilitados = 0;
        mostrar_ganador();
    }
}

static void actualizar_leds(void)
{
    //Encender LEDs del ganador
    if (ganador != 0) {
        mostrar_ganador();
        return;
    }

    //Apaga los dos jugadores antes de cambiar
    PORTB &= (uint8_t)~((1 << TRANSISTOR_J1) | (1 << TRANSISTOR_J2));

    if (jugador_led == 0) {
        mostrar_leds(decadas_J1);
        PORTB |= (1 << TRANSISTOR_J1);
        jugador_led = 1;
    } else {
        mostrar_leds(decadas_J2);
        PORTB |= (1 << TRANSISTOR_J2);
        jugador_led = 0;
    }

    for (volatile uint16_t pausa = 0; pausa < 200; pausa++) {
    }
}

static void mostrar_leds(uint8_t numero)
{
    uint8_t salida = 0;

    salida = numero & 0x0F;
    PORTB = (PORTB & 0b11110000) | salida;
}

static void mostrar_ganador(void)
{
    //Número del ganador en el display
    if (ganador == 1) {
        mostrar_numero(1);
        PORTB = 0b00011111;
    } else if (ganador == 2) {
        mostrar_numero(2);
        PORTB = 0b00101111;
    }
}

/****************************************/
// Interrupt routines

ISR(TIMER1_COMPA_vect)
{
    //No hacer nada sin conteo
    if (!conteo_activo) {
        return;
    }

    //Contar timer
    tiempo_timer++;

    //Esperar 1s
    if (tiempo_timer < 10) {
        return;
    }

    tiempo_timer = 0;

    //Bajar el numero del conteo
    if (numero_conteo > 0) {
        numero_conteo--;
        mostrar_numero(numero_conteo);
    }

    //0, habilitar a todos
    if (numero_conteo == 0) {
        conteo_activo = 0;
        jugadores_habilitados = 1;
    }
}
